# IBM Granite via watsonx.ai
def call_granite(prompt):
    token = get_ibm_iam_token()
    payload = {
        "model_id": "ibm/granite-13b-instruct-v2",
        "input": prompt,
        "parameters": {"max_new_tokens": 2000}
    }
    response = requests.post(
        watsonx_url, headers=auth_headers, json=payload
    )
    return response["results"][0]["generated_text"]
